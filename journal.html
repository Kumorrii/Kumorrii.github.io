	def FizzBuzz():
		'''Print out 1 to 100 but every multiple of 3 print Fizz, and every multiple of 5 print Buzz'''

				print("FizzBuzz")
						
			elif(num%3 == 0):
				print("Fizz")
					
			elif(num%5 == 0):
